\newcommand\descriptiontype{}
\def\descriptiontype[#1](#2)(#3)(#4:#5){%
    %\descriptiontype(right of which node)(Descriptive Name)(Type)(dx,dy)
    \node (name)[rectangle split, rectangle split parts=2, draw,text width = 5cm, minimum height=5cm, align = center, fill= cyan!80!blue,  #1 ]  at (#4,#5)
    {\bfseries #2
     \nodepart{two}
        \itshape  #3};
}



\newcommand\dataobj{} % just for safety
\def\dataobj[#1](#2)(#3)(#4)(#5)(#6)(#7){%
    
  % Synopsis
  % \dataobj[flag](descriptive name)(Type)(short description)(format)(size)(action)
    \ifx\relax#7\relax
    % if #7 is empty, do nothing
    \else
    % if #7 is not empty, create the action node
    \node(action)[actions]{#7};
    \fi
    \descriptiontype[right= of action](#2)(#3)(0:0)
    \node(description)[descriptions, right=2cm of name] { \small\itshape #4};
    \node(format)[formatsize, right=2cm of description] { \itshape #5};
    \node(size)[formatsize, right=2cm of format] { \itshape #6};
    \draw[-{Stealth[length=5mm]}] (name) -- (description);
    \draw[-{Stealth[length=5mm]}] (description) -- (format);
    \draw[-{Stealth[length=5mm]}] (format) -- (size);
    \ifx\relax#1\relax
    % if #1 is empty, do nothing
    \else
    % if #1 is not empty, create the callout node
    \node[rectangle callout, fill=red!50, draw, callout absolute pointer=(description.north), above=0.5cm of description, overlay]{\bfseries #1};
    \fi
    
    

    \\
}
